    <!-- Scripts simples -->
    <script>
        const products = [
            { id: 1, name: "Camiseta Serafín", price: 45.00, icon: "fa-tshirt" },
            { id: 2, name: "Gorra Sagrada", price: 30.00, icon: "fa-hat-wizard" },
            { id: 3, name: "Hoodie Arcángel", price: 85.00, icon: "fa-user-secret" },
            { id: 4, name: "Amuleto Oro", price: 120.00, icon: "fa-ankh" },
            { id: 5, name: "Túnica Noche", price: 55.00, icon: "fa-person-rays" },
            { id: 6, name: "Bolsa Ritual", price: 40.00, icon: "fa-sack-dollar" }
        ];

        const productContainer = document.querySelector('#catalogue .grid');
        const cartItemsDiv = document.getElementById('cart-items');
        const cartTotalSpan = document.getElementById('cart-total');
        const cartCountSpan = document.getElementById('cart-count');
        const cartModal = document.getElementById('cart-modal');
        
        let cart = [];

        // Renderizar productos
        productContainer.innerHTML = products.map(p => `
            <div class="product-cell">
                <div class="product-img-box">
                    <i class="fa-solid ${p.icon} text-4xl text-[#bf9b30]"></i>
                </div>
                <strong class="block text-[#bf9b30] font-serif text-lg">${p.name}</strong>
                <div class="text-xs text-gray-500 mb-2 font-mono">REF-${1000 + p.id}</div>
                <div class="text-white font-bold mb-2">$${p.price.toFixed(2)}</div>
                <button onclick="addToCart(${p.id})" class="btn-retro text-xs">
                    [ AGREGAR ]
                </button>
            </div>
        `).join('');

        function toggleCart() {
            cartModal.classList.toggle('hidden');
            renderCart();
        }

        function addToCart(id) {
            const prod = products.find(p => p.id === id);
            cart.push(prod);
            cartCountSpan.innerText = cart.length;
            alert(">> ITEM AÑADIDO AL INVENTARIO <<");
        }

        function removeFromCart(index) {
            cart.splice(index, 1);
            cartCountSpan.innerText = cart.length;
            renderCart();
        }

        function renderCart() {
            if (cart.length === 0) {
                cartItemsDiv.innerHTML = "<div class='text-center py-4'>... VACÍO ...</div>";
                cartTotalSpan.innerText = "$0.00";
                return;
            }
            
            let total = 0;
            cartItemsDiv.innerHTML = cart.map((item, idx) => {
                total += item.price;
                return `
                <div class="border-b border-gray-800 py-2 flex justify-between">
                    <span>${item.name}</span>
                    <div>
                        <span>$${item.price}</span>
                        <button onclick="removeFromCart(${idx})" class="text-red-500 ml-2">[x]</button>
                    </div>
                </div>
            `}).join('');
            
            cartTotalSpan.innerText = "$" + total.toFixed(2);
        }

        function checkout() {
            if(cart.length === 0) {
                alert("Error: El carrito está vacío.");
                return;
            }
            alert("Procesando pago... Conectando al servidor...");
            setTimeout(() => {
                cart = [];
                cartCountSpan.innerText = 0;
                toggleCart();
                alert("¡GRACIAS POR TU COMPRA!");
            }, 1500);
        }
    </script>
